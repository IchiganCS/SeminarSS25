def sort[T](list: List[T])(implicit ord: Ordering[T]): List[T] = {
  list match {
    case Nil => Nil
    case x :: xs =>
      // partition list based on pivot element x
      val (lo, hi) = xs.partition(ord.lt(_, x))  //*\label{srcline:algcmp}
      sort(lo) ++ (x :: sort(hi))
  }
}
\end{lstlisting}
